00032 static __attribute__((noinline)) void fill8(uint8_t* bgn, uint8_t* end) {
00033   while (bgn < end) *bgn++ = 0X55;

00070 /** Fill stacks with 0X55.*/
00071 static void nilFillStacks() {
00072   thread_config_t *tcp = nil_thd_configs;
00073   
00074   while (tcp->wap) {
00075     fill8(tcp->wap, tcp->wap + tcp->size);
00076     tcp++;
00077   }
00078   // fill heap/idle stack
00079   fill8(heapEnd(), (uint8_t*)&tcp - 16);
00080 }
