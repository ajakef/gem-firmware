00056  * Start periodic timer 1 signals.
00057  *
00058  * @param[in] microseconds The period in microseconds between signals.
00059  * @note The maximum value for microseconds is 4,194,304 for a 16 MHz CPU.
00060  */
00061  void nilTimer1Start(uint32_t microseconds){
00062  // CPU cycles per time period.
00063  long cycles = (F_CPU / 1000000) * microseconds;
00064 
00065   nilTimer1Stop();
00066   
00067   uint8_t tshift;
00068   if (cycles <= 0X10000) {
00069     // no prescale, CTC mode
00070     TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS10);
00071     tshift = 0;
00072   } else if (cycles <= 0X10000*8L) {
00073     // prescale 8, CTC mode
00074     TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS11);
00075     tshift = 3;
00076   } else if (cycles <= 0X10000*64L) {
00077     // prescale 64, CTC mode
00078     TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS11) | (1 << CS10);
00079     tshift = 6;
00080   } else if (cycles <= 0X10000*256L) {
00081     // prescale 256, CTC mode
00082     TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS12);
00083     tshift = 8;
00084   } else {
00085     // prescale 1024, CTC mode
00086     TCCR1B = (1 << WGM13) | (1 << WGM12) | (1 << CS12) | (1 << CS10);
00087     tshift = 10;
00088     // microseconds too large, use max time.
00089     if (cycles > 0X10000*1024L) cycles = 0X10000*1024L;
00090   }
00091   // divide by prescaler
00092   cycles >>= tshift;
00093   
00094   // set TOP for timer reset
00095   nilSysLock();
00096   ICR1 = cycles - 1;
00097   TCNT1 = 0;
00098   // clear pending interrupt
00099   TIFR1 |= (1 << OCF1B);
00100   // enable compare B interrupt
00101   TIMSK1 = (1 << OCIE1B);
00102   nilSysUnlock();
00103 
00104  }
