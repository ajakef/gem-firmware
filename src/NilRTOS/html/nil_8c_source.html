00059 /**
00060  * @brief   Initializes the kernel.
00061  * @details Initializes the kernel structures, the current instructions flow
00062  *          becomes the idle thread upon return. The idle thread must not
00063  *          invoke any kernel primitive able to change state to not runnable.
00064  *
00065  * @special
00066  */
00067 void nilSysInit(void) {
00068   thread_ref_t tr;
00069   const thread_config_t *tcp;
00070 
00071   /* Port layer initialization.*/
00072   port_init();
00073 
00074   /* Iterates through the list of defined threads.*/
00075   for (tr = &nil.threads[0], tcp = nil_thd_configs;
00076   #if WHG_MOD
00077        tr < nil.idlep;
00078 #else  /* WHG_MOD */
00079        tr < &nil.threads[NIL_CFG_NUM_THREADS];
00080 #endif  /* WHG_MOD */
00081        tr++, tcp++) {
00082     tr->state = NIL_THD_READY;
00083     tr->timeout = 0;
00084 
00085     /* Port dependent thread initialization.*/
00086     SETUP_CONTEXT(tr, tcp->wap, tcp->size, tcp->funcp, tcp->arg);
00087 
00088     /* Initialization hook.*/
00089 #if defined(NIL_CFG_THREAD_EXT_INIT_HOOK)
00090     NIL_CFG_THREAD_EXT_INIT_HOOK(tr);
00091 #endif
00092   }
00093 
00094   /* Runs the highest priority thread, the current one becomes the null
00095      thread.*/
00096   nil.current = nil.next = nil.threads;
00097 #if WHG_MOD
00098   port_switch(nil.threads, nil.idlep);
00099 #else  /* WHG_MOD */
00100   port_switch(nil.threads, &nil.threads[NIL_CFG_NUM_THREADS]);
00101 #endif  /* WHG_MOD */
00102 
00103   /* Interrupts enabled for the idle thread.*/
00104   nilSysEnable();
00105 }

00164       if (tr->timeout == 0) {
00165         /* Timeout on semaphores requires a special handling because the

00275   nilDbgAssert(otr != nil.idlep,
00276                 "nilSchGoSleepTimeoutS(), #1", "idle cannot sleep");
00277 #else  /* WHG_MOD */
00278   nilDbgAssert(otr != &nil.threads[NIL_CFG_NUM_THREADS],
00279                "nilSchGoSleepTimeoutS(), #1", "idle cannot sleep");
00280 #endif  /* WHG_MOD */
00281 
00282 
00283   /* Storing the wait object for the current thread.*/
00284   otr->state = newstate;
00285 
00286 #if NIL_CFG_TIMEDELTA > 0
00287   if (timeout != TIME_INFINITE) {
00288     systime_t time;
00289 
00290     /* TIMEDELTA makes sure to have enough time to reprogram the timer
00291        before the free-running timer counter reaches the selected timeout.*/
00292     if (timeout < NIL_CFG_TIMEDELTA)
00293       timeout = NIL_CFG_TIMEDELTA;
00294 
00295     time = nilTimeNowI() + timeout;
00296     if (nilTimeIsWithin(time, nil.lasttime, nil.nexttime)) {
00297       port_timer_set_alarm(time);
00298       nil.nexttime = time;
00299     }
00300 
00301     /* Timeout settings.*/
00302     otr->timeout = time - nil.lasttime;
00303   }
00304 #else
00305 
00306   /* Timeout settings.*/
00307   otr->timeout = timeout;

00309 
00310   /* Scanning the whole threads array.*/
00311   ntr = nil.threads;
00312   while (true) {

00317 #if WHG_MOD
00318       if (ntr == nil.idlep) {
00319 #else  /* WHG_MOD */
00320       if (ntr == &nil.threads[NIL_CFG_NUM_THREADS]) {
00321 #endif  /* WHG_MOD */
00322         NIL_CFG_IDLE_ENTER_HOOK();

00330     ntr++;
00331 #if WHG_MOD
00332     nilDbgAssert(ntr <= nil.idlep,
00333                  "nilSchGoSleepTimeoutS(), #2", "pointer out of range");
00334 #else  /* WHG_MOD */
00335     nilDbgAssert(ntr <= &nil.threads[NIL_CFG_NUM_THREADS],
